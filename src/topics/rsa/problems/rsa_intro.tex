 \textbf{RSA}: Given two large primes, $p$ and $q$, and a message $x$,
 \begin{align*}
     N = pq \\
     E(x) = x^e \text{mod} N \\
     D(x) = x^d \text{mod} N
 \end{align*} \\
 where $e$ is relatively prime to $(p - 1)(q - 1)$, and $ed = 1$. The pair $(N, e)$
 is the recipient's \textbf{public key}, and $d$ is the recipient's \textbf{private key}.
 The sender sends $E(x)$ to the recipient, and the recipient uses $D(x)$ to recover the original message.
 \end{minipage}}
