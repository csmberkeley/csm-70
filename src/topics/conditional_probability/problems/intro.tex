\vspace{2 mm}
\textbf{Bayes' Rule} 
\[ \P[A|B] = \frac{\P[B|A]P[A]}{\P[B]} \]

\textbf{Total Probability Rule}
\[ \P[B] = \P[A \cap B] + \P[ \bar{A} \cap B] = \P[B|A] * \P[A] + 
\P[B|\bar{A}]*(1 - \P[A])\]

\textbf{Independence} \newline 
Two events $A$, $B$ in the same probability space are independent if 
\[\P[A \cap B] = \P[A] * \P[B] \]
\\
